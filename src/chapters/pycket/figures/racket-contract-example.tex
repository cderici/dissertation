#lang racket

;; fibonacci with contract
(define/contract (fib n)
  (-> exact-nonnegative-integer?
      exact-nonnegative-integer?)
  (match n
    [0 0]
    [1 1]
    [_ (+ (fib (- n 1)) (fib (- n 2)))]))

(module+ main
  (displayln (for/list ([i (in-range 10)])
               (fib i))))
