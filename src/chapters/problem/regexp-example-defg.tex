#lang racket/base

(define l (make-string 4000 #\a))
(define r (make-string 4000 #\p))
(define str-big (string-append l "defg" r))

(define reg-r (regexp "defg"))
(define iter 1000000)

(time
  (for ([i (in-range iter)])
    (regexp-match reg-r str-big)))
