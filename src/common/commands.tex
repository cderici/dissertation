    \begingroup
      \parindent \z@
      \rightskip \@pnumwidth
      \parfillskip -\rightskip
      \leavevmode
      \large\bfseries
      % actual text
      Part~\thepart\space#1\hfil\hb@xt@\@pnumwidth{\hss #2}%
      \par
      \addvspace{0.5em}
    \endgroup
  \fi}
